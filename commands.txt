
#NODEJS
    cinst nodejs-lts -- Instalar o NodeJS
    nom init -y -- package.json backend
    npm install express -- Criar projeto NodeJS Express
    node index.js -- Inicia servidor node
        #NODEMON
            npm install nodemon -D -- Instalar nodemon monitora alterações no NodeJS e o reinicializa sozinho ('-D': devdependecies)
            "start": "nodemon index.js" -- script para inicializar o nodemon (npm start)
        #KNEX
            npm install knex -- Instalar knex
            npm install sqlite3 -- Instala o BD SQLite
            npx knex init -- Inicializar knex (knexfile.js: conexões com bd)
            npx knex migrate:make create_table -- Criar migration (essa vai criar a tabela table)
            npx knex migrate:latest -- Rodar a migration
            npx knex migrate:rollback -- Desfaz ultima migrate:latest

#REACTJS
    npx create-react-app nomeprojeto -- Criar estrutura do ReactJS
